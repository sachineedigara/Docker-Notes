Docker secrets created by user and then using in docker stack


printf "abc" | docker secret create mysql_password -

printf "xyz" | docker secret create mysql_root_password -


version: '3.1'
services:
  wordpress:
    depends_on:
      - database
    deploy:
      replicas: 2
    image: wordpress:latest
    ports:
      - 31000:80
    secrets:
    - mysql_password
    environment:
      WORDPRESS_DB_HOST: database:3306
      WORDPRESS_DB_NAME: wp_db
      WORDPRESS_DB_USER: wordpress
      WORDPRESS_DB_PASSWORD_FILE: /run/secrets/mysql_password
  database:
    image: mysql:5.7
    secrets:
    - mysql_password
    - mysql_root_password
    environment:
      MYSQL_ROOT_PASSWORD_FILE: /run/secrets/mysql_root_password
      MYSQL_DATABASE: wp_db
      MYSQL_USER: wordpress
      MYSQL_PASSWORD_FILE: /run/secrets/mysql_password
    volumes:
     - db-vol:/var/lib/mysql
volumes:
  db-vol:
secrets:
  mysql_password:
    external: true
  mysql_root_password:
    external: true
